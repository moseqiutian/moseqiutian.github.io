3、String 对象实例方法


a.indexOf（Chr）：返回指定子字符串的位置，从左到右。找不到返回-1。 


实例
			var str="Hello world!"
			document.write(str.indexOf("Hello") + "br")
			document.write(str.indexOf("World") + "br ")
			document.write(str.indexOf("world"))
			
结果
				0

				-1

				6

b.charAt（idx）：返回指定位置处的字符 


实例
				var str="Hello world!"
				document.write(str.charAt(1))
			
结果
				e

二、Array对象


1、Array 对象实例方法


a.push() 方法可向数组的末尾添加一个或多个元素，并返回新的长度。


实例
				var arr = new Array(3)
				arr[0] = "George"
				arr[1] = "John"
				arr[2] = "Thomas"
				
				document.write(arr + "br")
				document.write(arr.push("James") + "br")
				document.write(arr)
			
结果
				George,John,Thomas

				4

				George,John,Thomas,James

b.pop() 方法用于删除并返回数组的最后一个元素。


实例
				var arr = new Array(3)
				arr[0] = "George"
				arr[1] = "John"
				arr[2] = "Thomas"
				
				document.write(arr)
				document.write("br ")
				document.write(arr.pop())
				document.write("br")
				document.write(arr)
			
结果
				George,John,Thomas

				Thomas

				George,John

c.slice() 方法可从已有的数组中返回选定的元素。


实例1
				var arr = new Array(3)
				arr[0] = "George"
				arr[1] = "John"
				arr[2] = "Thomas"
				
                document.write(arr + "br ")
                document.write(arr.slice(1) + "br")
                 document.write(arr)
			
结果
				George,John,Thomas

				John,Thomas

				George,John,Thomas

			
实例2
				var arr = new Array(6)
				arr[0] = "George"
				arr[1] = "John"
				arr[2] = "Thomas"
				arr[3] = "James"
				arr[4] = "Adrew"
				arr[5] = "Martin"
				
				document.write(arr + "br ")
				document.write(arr.slice(2,4) + "br ")
				document.write(arr)
			
结果
				George,John,Thomas,James,Adrew,Martin

				Thomas,James

				George,John,Thomas,James,Adrew,Martin

d.splice() 方法向/从数组中添加/删除项目，然后返回被删除的项目。


实例1,在本例中，我们将创建一个新数组，并向其添加一个元素：
				
				var arr = new Array(6)
				arr[0] = "George"
				arr[1] = "John"
				arr[2] = "Thomas"
				arr[3] = "James"
				arr[4] = "Adrew"
				arr[5] = "Martin"
				
                document.write(arr + "br ")
                arr.splice(2,0,"William")
                document.write(arr + "br")
			
结果
				George,John,Thomas,James,Adrew,Martin

				George,John,William,Thomas,James,Adrew,Martin

			
实例2
				var arr = new Array(6)
				arr[0] = "George"
				arr[1] = "John"
				arr[2] = "Thomas"
				arr[3] = "James"
				arr[4] = "Adrew"
				arr[5] = "Martin"
				
                document.write(arr + "br ")
                arr.splice(2,1,"William")
                document.write(arr)
			
结果
				George,John,Thomas,James,Adrew,Martin

				George,John,William,James,Adrew,Martin
